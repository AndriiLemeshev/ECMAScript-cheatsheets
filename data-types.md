```js
// Primitive Data Types

// Numbers (both integers and floating-point)
123; // Integer
3.14; // Floating-point
1e6; // Scientific notation (1 * 10^6 = 1,000,000)

// Strings (sequences of characters)
"Hello"; // Double quotes
'World'; // Single quotes
`Template literals`; // Backticks (allow for string interpolation)

// Booleans (true or false values)
true;
false;

// Null (intentional absence of a value)
null;

// Undefined (variable declared but not assigned a value)
let myVariable; // myVariable is undefined
undefined; // Explicitly undefined

// Symbol (unique and immutable values, often used as object property keys)
Symbol('mySymbol');

// BigInt (for arbitrarily large integers)
1234567890123456789012345678901234567890n; // Note the 'n' suffix


// Non-primitive Data Type: Object

// Objects (collections of key-value pairs)
const myObject = {
  name: "John",
  age: 30,
  city: "New York"
};

// Arrays (ordered lists of values)
const myArray = [1, 2, 3, "four"];
```
